shell脚本之根据软链接,找到真实路径
#!/bin/bash
[[ $1 =~ ^/  ]] && a=$1 || a=`pwd`/$1
while [ -h $a ]
do
 
   b=`ls -ld $a|awk '{print $NF}'`
   c=`ls -ld $a|awk '{print $(NF-2)}'`
   [[ $b =~ ^/ ]] && a=$b  || a=`dirname $c`/$b
done
echo $a
